// React component example
const VoiceSelector = () => {
  const [voices, setVoices] = useState([]);

  useEffect(() => {
    fetch('/api/voice-cloning/voices')
      .then(res => res.json())
      .then(setVoices);
  }, []);

  return (
    <select>
      {voices.map(voice => (
        <option key={voice.id} value={voice.audioPath}>
          {voice.name} {voice.isPublic ? 'ğŸŒ' : 'ğŸ”’'}
        </option>
      ))}
    </select>
  );
}
